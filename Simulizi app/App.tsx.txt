
import React, { useState, useEffect } from 'react';
import { Story, User, Transaction, Episode, Language, PayoutRequest } from './types';
import { INITIAL_STORIES, PRICING } from './constants';
import { ReaderView } from './components/ReaderView';
import { WriterDashboard } from './components/WriterDashboard';
import { WriterRegistration } from './components/WriterRegistration';
import { AdminPanel } from './components/AdminPanel';
import { geminiService } from './services/geminiService';
import { translations } from './translations';

const App: React.FC = () => {
  const [view, setView] = useState<'reader' | 'writer' | 'admin'>('reader');
  const [lang, setLang] = useState<Language>('en');
  const [stories, setStories] = useState<Story[]>([]);
  const [user, setUser] = useState<User | null>(null);
  const [transactions, setTransactions] = useState<Transaction[]>([]);
  const [unlockedEpisodes, setUnlockedEpisodes] = useState<string[]>([]);
  const [payoutRequests, setPayoutRequests] = useState<PayoutRequest[]>([]);
  
  const [showContactModal, setShowContactModal] = useState(false);
  const [showAuthModal, setShowAuthModal] = useState(false);
  const [showTopUpModal, setShowTopUpModal] = useState(false);
  const [topUpAmount, setTopUpAmount] = useState(1000);

  const t = translations[lang];

  useEffect(() => {
    const savedStories = localStorage.getItem('stories');
    let currentStories: Story[];
    if (savedStories) {
      currentStories = JSON.parse(savedStories);
    } else {
      currentStories = INITIAL_STORIES;
    }
    setStories(currentStories);

    const savedTx = localStorage.getItem('transactions');
    if (savedTx) setTransactions(JSON.parse(savedTx));

    const savedUnlocked = localStorage.getItem('unlocked');
    if (savedUnlocked) setUnlockedEpisodes(JSON.parse(savedUnlocked));

    const savedPayouts = localStorage.getItem('payouts');
    if (savedPayouts) setPayoutRequests(JSON.parse(savedPayouts));

    const savedUser = localStorage.getItem('currentUser');
    if (savedUser) {
      setUser(JSON.parse(savedUser));
    } else {
      setShowAuthModal(true);
    }

    const savedLang = localStorage.getItem('lang') as Language;
    if (savedLang) setLang(savedLang);
  }, []);

  useEffect(() => { localStorage.setItem('stories', JSON.stringify(stories)); }, [stories]);
  useEffect(() => { localStorage.setItem('transactions', JSON.stringify(transactions)); }, [transactions]);
  useEffect(() => { localStorage.setItem('unlocked', JSON.stringify(unlockedEpisodes)); }, [unlockedEpisodes]);
  useEffect(() => { localStorage.setItem('payouts', JSON.stringify(payoutRequests)); }, [payoutRequests]);
  useEffect(() => { if (user) localStorage.setItem('currentUser', JSON.stringify(user)); }, [user]);
  useEffect(() => { localStorage.setItem('lang', lang); }, [lang]);

  const calculateWriterBalance = (writerId: string) => {
    const totalEarned = transactions
      .filter(tx => stories.find(s => s.id === tx.storyId)?.authorId === writerId)
      .reduce((acc, curr) => acc + curr.net, 0);
    
    const totalWithdrawn = payoutRequests
      .filter(p => p.writerId === writerId && p.status === 'completed')
      .reduce((acc, curr) => acc + curr.amount, 0);

    return totalEarned - totalWithdrawn;
  };

  const handleRegisterWriter = (name: string, bio: string) => {
    if (!user) return;
    const updatedUser: User = { ...user, name, role: 'writer', balance: 0 };
    setUser(updatedUser);
  };

  const handleOnboarding = (name: string, role: 'reader' | 'writer') => {
    const newUser: User = { id: `u_${Date.now()}`, name, role, balance: role === 'reader' ? 1000 : 0 };
    setUser(newUser);
    setShowAuthModal(false);
    setView(role === 'writer' ? 'writer' : 'reader');
  };

  const handlePay = (storyId: string, episodeId: string) => {
    if (!user) return;
    if (user.balance < PRICING.EPISODE_COST_TZS) {
      alert(t.insufficientFunds);
      setShowTopUpModal(true);
      return;
    }
    const newTx: Transaction = {
      id: `tx_${Date.now()}`, userId: user.id, storyId, episodeId,
      amount: PRICING.EPISODE_COST_TZS, fee: PRICING.SERVICE_FEE_TZS, net: PRICING.WRITER_NET_TZS,
      timestamp: Date.now()
    };
    setTransactions([...transactions, newTx]);
    setUnlockedEpisodes([...unlockedEpisodes, episodeId]);
    setUser({ ...user, balance: user.balance - PRICING.EPISODE_COST_TZS });
    alert(lang === 'en' ? "Payment Successful! Episode Unlocked." : "Malipo Yamefanikiwa! Kipindi Kimefunguliwa.");
  };

  const handleTopUp = () => {
    if (!user) return;
    setUser({ ...user, balance: user.balance + topUpAmount });
    setShowTopUpModal(false);
    alert(t.paymentSuccess);
  };

  const handleRequestPayout = (payout: PayoutRequest) => {
    setPayoutRequests([...payoutRequests, payout]);
  };

  const handleCompletePayout = (id: string) => {
    setPayoutRequests(payoutRequests.map(p => p.id === id ? { ...p, status: 'completed' } : p));
  };

  const toggleLang = () => setLang(prev => (prev === 'en' ? 'sw' : 'en'));

  const currentUserBalance = user ? (user.role === 'writer' ? calculateWriterBalance(user.id) : user.balance) : 0;

  return (
    <div className="min-h-screen bg-slate-50 text-slate-900">
      {/* Onboarding Overlay */}
      {showAuthModal && (
        <div className="fixed inset-0 z-[200] bg-indigo-900/90 backdrop-blur-md flex items-center justify-center p-4">
          <div className="bg-white rounded-[40px] shadow-2xl p-8 max-w-md w-full animate-scale-up">
            <div className="text-center mb-8">
              <div className="w-20 h-20 bg-indigo-600 rounded-3xl flex items-center justify-center text-white font-bold text-4xl mx-auto mb-6 shadow-xl shadow-indigo-200">S</div>
              <h2 className="text-3xl font-serif font-bold text-slate-900">{t.welcome}</h2>
              <p className="text-slate-500 mt-2">{t.onboardingDesc}</p>
            </div>
            <form onSubmit={(e) => {
              e.preventDefault();
              const formData = new FormData(e.currentTarget);
              handleOnboarding(formData.get('name') as string, formData.get('role') as any);
            }} className="space-y-6">
              <div>
                <label className="block text-xs font-black text-slate-400 uppercase tracking-widest mb-2">{t.displayName}</label>
                <input name="name" required type="text" className="w-full px-5 py-4 rounded-2xl border border-slate-200 focus:ring-4 focus:ring-indigo-500/10 focus:border-indigo-600 outline-none transition-all text-lg font-medium" placeholder="e.g. Salim Ali" />
              </div>
              <div>
                <label className="block text-xs font-black text-slate-400 uppercase tracking-widest mb-2">{t.chooseRole}</label>
                <div className="grid grid-cols-2 gap-4">
                  <label className="relative flex flex-col items-center p-4 rounded-2xl border-2 border-slate-100 cursor-pointer hover:bg-indigo-50 has-[:checked]:border-indigo-600 has-[:checked]:bg-indigo-50 transition-all">
                    <input type="radio" name="role" value="reader" defaultChecked className="hidden" />
                    <span className="text-2xl mb-1">üìñ</span>
                    <span className="font-bold text-sm text-slate-700">{t.readerRole}</span>
                  </label>
                  <label className="relative flex flex-col items-center p-4 rounded-2xl border-2 border-slate-100 cursor-pointer hover:bg-indigo-50 has-[:checked]:border-indigo-600 has-[:checked]:bg-indigo-50 transition-all">
                    <input type="radio" name="role" value="writer" className="hidden" />
                    <span className="text-2xl mb-1">‚úçÔ∏è</span>
                    <span className="font-bold text-sm text-slate-700">{t.writerRole}</span>
                  </label>
                </div>
              </div>
              <button type="submit" className="w-full py-5 bg-indigo-600 text-white rounded-3xl font-black text-lg shadow-2xl shadow-indigo-200 hover:bg-indigo-700 hover:scale-[1.02] active:scale-[0.98] transition-all">{t.createAccount}</button>
            </form>
          </div>
        </div>
      )}

      {/* Top Up Modal */}
      {showTopUpModal && (
        <div className="fixed inset-0 z-[200] bg-slate-900/60 backdrop-blur-sm flex items-center justify-center p-4">
          <div className="bg-white rounded-[32px] shadow-2xl p-8 max-w-sm w-full animate-scale-up border border-slate-100">
            <h3 className="text-2xl font-serif font-bold text-slate-900 mb-2">{t.topUp}</h3>
            <p className="text-sm text-slate-500 mb-6">{t.selectAmount} (TZS)</p>
            <div className="grid grid-cols-2 gap-3 mb-8">
              {[1000, 2000, 5000, 10000].map(amt => (
                <button key={amt} onClick={() => setTopUpAmount(amt)} className={`py-3 rounded-2xl font-bold border-2 transition-all ${topUpAmount === amt ? 'border-indigo-600 bg-indigo-50 text-indigo-600' : 'border-slate-100 text-slate-500 hover:bg-slate-50'}`}>{amt.toLocaleString()}</button>
              ))}
            </div>
            <div className="mb-8">
              <label className="block text-[10px] font-black text-slate-400 uppercase tracking-widest mb-3">{t.payWith}</label>
              <div className="flex gap-4">
                <div className="w-12 h-8 bg-slate-100 rounded flex items-center justify-center text-[10px] font-bold text-slate-400">M-PESA</div>
                <div className="w-12 h-8 bg-slate-100 rounded flex items-center justify-center text-[10px] font-bold text-slate-400">TIGO</div>
              </div>
            </div>
            <div className="flex gap-3">
              <button onClick={() => setShowTopUpModal(false)} className="flex-1 py-4 border border-slate-200 rounded-2xl font-bold text-slate-500 hover:bg-slate-50">{t.close}</button>
              <button onClick={handleTopUp} className="flex-1 py-4 bg-indigo-600 text-white rounded-2xl font-bold hover:bg-indigo-700 shadow-lg shadow-indigo-100">{t.confirmPayment}</button>
            </div>
          </div>
        </div>
      )}

      {/* Navigation */}
      <nav className="bg-white border-b border-slate-200 sticky top-0 z-50 shadow-sm">
        <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8">
          <div className="flex justify-between h-20 items-center">
            <div className="flex items-center gap-3">
              <div className="w-12 h-12 bg-indigo-600 rounded-xl flex items-center justify-center text-white font-black text-2xl shadow-lg shadow-indigo-100">S</div>
              <h1 className="text-xl md:text-2xl font-serif font-bold tracking-tight">Simulizi<span className="text-indigo-600"> na Hadithi</span></h1>
            </div>
            <div className="hidden lg:flex items-center space-x-2">
              <button onClick={() => setView('reader')} className={`px-5 py-2.5 rounded-full text-sm font-black uppercase tracking-widest transition ${view === 'reader' ? 'bg-indigo-600 text-white shadow-lg' : 'text-slate-500 hover:text-indigo-600'}`}>{t.explore}</button>
              <button onClick={() => setView('writer')} className={`px-5 py-2.5 rounded-full text-sm font-black uppercase tracking-widest transition ${view === 'writer' ? 'bg-indigo-600 text-white shadow-lg' : 'text-slate-500 hover:text-indigo-600'}`}>{t.writerStudio}</button>
              <button onClick={() => setView('admin')} className={`px-5 py-2.5 rounded-full text-sm font-black uppercase tracking-widest transition ${view === 'admin' ? 'bg-indigo-600 text-white shadow-lg' : 'text-slate-500 hover:text-indigo-600'}`}>{t.admin}</button>
            </div>
            <div className="flex items-center gap-4">
              <button onClick={toggleLang} className="hidden md:flex items-center gap-2 px-4 py-2 rounded-xl bg-slate-50 text-[10px] font-black uppercase tracking-widest border border-slate-200 text-slate-600 hover:bg-slate-100 transition">{lang === 'en' ? 'SW' : 'EN'}</button>
              {user && (
                <div className="flex items-center gap-4 pl-4 border-l border-slate-100">
                  <div className="text-right hidden sm:block">
                    <p className="text-[10px] text-slate-400 font-black uppercase tracking-[0.2em]">{user.role}</p>
                    <div className="flex items-center gap-2">
                      <p className="text-sm font-black text-slate-900">TZS {currentUserBalance.toLocaleString()}</p>
                      {user.role === 'reader' && (
                        <button onClick={() => setShowTopUpModal(true)} className="w-5 h-5 rounded-full bg-indigo-600 text-white flex items-center justify-center hover:scale-110 transition"><svg className="w-3 h-3" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth={3} d="M12 4v16m8-8H4" /></svg></button>
                      )}
                    </div>
                  </div>
                  <div className="w-12 h-12 rounded-2xl bg-slate-100 overflow-hidden ring-4 ring-indigo-50 border border-slate-200 shrink-0"><img src={`https://api.dicebear.com/7.x/avataaars/svg?seed=${user.name}`} alt="avatar" /></div>
                </div>
              )}
            </div>
          </div>
        </div>
      </nav>

      <main className="max-w-7xl mx-auto px-4 py-12 sm:px-6 lg:px-8">
        {user && (
          <>
            {view === 'reader' && <ReaderView stories={stories.filter(s => s.status === 'approved')} onPay={handlePay} unlockedEpisodes={unlockedEpisodes} lang={lang} />}
            {view === 'writer' && (
              user.role === 'writer' ? (
                <WriterDashboard 
                  stories={stories.filter(s => s.authorId === user.id)}
                  currentUser={user}
                  onAddStory={(s) => setStories([...stories, s])}
                  onUpdateStory={(s) => setStories(stories.map(st => st.id === s.id ? s : st))}
                  lang={lang}
                  balance={currentUserBalance}
                  onWithdraw={handleRequestPayout}
                />
              ) : <WriterRegistration onRegister={handleRegisterWriter} lang={lang} />
            )}
            {view === 'admin' && (
              <AdminPanel 
                stories={stories} 
                onUpdateStatus={(id, status) => setStories(stories.map(s => s.id === id ? { ...s, status } : s))}
                lang={lang}
                payoutRequests={payoutRequests}
                onCompletePayout={handleCompletePayout}
              />
            )}
          </>
        )}
      </main>

      <footer className="bg-white border-t border-slate-200 py-16 mb-24 md:mb-0">
        <div className="max-w-7xl mx-auto px-4 text-center">
          <div className="flex items-center justify-center gap-3 mb-6"><div className="w-10 h-10 bg-indigo-600 rounded-xl flex items-center justify-center text-white font-black text-xl shadow-lg shadow-indigo-100">S</div><h2 className="text-2xl font-serif font-bold">Simulizi na Hadithi</h2></div>
          <p className="text-slate-400 text-sm max-w-lg mx-auto mb-10 leading-relaxed">{lang === 'en' ? "Bridging the gap between brilliant storytellers and curious listeners." : "Kuunganisha wasimulizi mahiri na wasikilizaji wadadisi."}</p>
          <div className="flex justify-center gap-12 text-slate-400 text-xs font-black uppercase tracking-widest mb-10">
            <button onClick={() => setShowContactModal(true)} className="hover:text-indigo-600 transition">{t.contactUs}</button>
            <button onClick={() => setShowAuthModal(true)} className="hover:text-indigo-600 transition">{t.switchAccount}</button>
          </div>
          <p className="text-[10px] text-slate-400 font-black tracking-[0.3em] uppercase">&copy; 2025 Simulizi na Hadithi.</p>
        </div>
      </footer>
    </div>
  );
};

export default App;
