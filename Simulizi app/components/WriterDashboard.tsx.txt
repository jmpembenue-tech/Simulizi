
import React, { useState, useRef } from 'react';
import { Story, Episode, User, Language, PayoutRequest } from '../types';
import { PRICING } from '../constants';
import { geminiService } from '../services/geminiService';
import { translations } from '../translations';

interface WriterDashboardProps {
  stories: Story[];
  currentUser: User;
  onAddStory: (story: Story) => void;
  onUpdateStory: (story: Story) => void;
  lang: Language;
  balance: number;
  onWithdraw: (payout: PayoutRequest) => void;
}

export const WriterDashboard: React.FC<WriterDashboardProps> = ({ stories, currentUser, onAddStory, onUpdateStory, lang, balance, onWithdraw }) => {
  const [isAddingProject, setIsAddingProject] = useState(false);
  const [showWithdrawModal, setShowWithdrawModal] = useState(false);
  const [activeStoryForUpload, setActiveStoryForUpload] = useState<Story | null>(null);
  const [isGeneratingAudio, setIsGeneratingAudio] = useState<string | null>(null);
  const [isUploading, setIsUploading] = useState(false);
  
  const [withdrawForm, setWithdrawForm] = useState({ amount: 2000, network: 'M-Pesa', phone: '' });
  const [projectForm, setProjectForm] = useState({ title: '', description: '', type: 'text' as 'text' | 'audio', isPaid: true });
  const [episodeForm, setEpisodeForm] = useState({ title: '', file: null as File | null });

  const t = translations[lang];

  const handleWithdraw = () => {
    if (withdrawForm.amount < 2000) { alert(t.minWithdrawal); return; }
    if (withdrawForm.amount > balance) { alert(t.insufficientFunds); return; }
    if (!withdrawForm.phone.trim()) { alert(t.paymentDetails); return; }

    const newPayout: PayoutRequest = {
      id: `p_${Date.now()}`, writerId: currentUser.id, writerName: currentUser.name,
      amount: withdrawForm.amount, paymentMethod: withdrawForm.network,
      phoneNumber: withdrawForm.phone, status: 'pending', timestamp: Date.now()
    };
    onWithdraw(newPayout);
    setShowWithdrawModal(false);
    alert(t.requestSent);
  };

  const handleCreateStory = () => {
    if (!projectForm.title.trim()) return;
    const newStory: Story = {
      id: `s_${Date.now()}`, title: projectForm.title, authorId: currentUser.id, authorName: currentUser.name,
      description: projectForm.description, coverImage: `https://api.dicebear.com/7.x/initials/svg?seed=${projectForm.title}&backgroundColor=4f46e5`,
      isPaid: projectForm.isPaid, status: 'pending', type: projectForm.type, episodes: [], createdAt: Date.now()
    };
    onAddStory(newStory);
    setIsAddingProject(false);
    setProjectForm({ title: '', description: '', type: 'text', isPaid: true });
  };

  const readFileAsync = (file: File, asText: boolean): Promise<string> => {
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.onload = () => resolve(reader.result as string);
      reader.onerror = reject;
      if (asText) reader.readAsText(file); else reader.readAsDataURL(file);
    });
  };

  const handleUploadEpisode = async () => {
    if (!activeStoryForUpload || !episodeForm.title.trim() || !episodeForm.file) { alert(t.noFileSelected); return; }
    setIsUploading(true);
    try {
      const isText = activeStoryForUpload.type === 'text';
      const content = await readFileAsync(episodeForm.file, isText);
      const newEpisode: Episode = { 
        id: `e_${Date.now()}`, 
        storyId: activeStoryForUpload.id, 
        title: episodeForm.title, 
        content, 
        order: activeStoryForUpload.episodes.length + 1 
      };
      onUpdateStory({ 
        ...activeStoryForUpload, 
        episodes: [...activeStoryForUpload.episodes, newEpisode], 
        status: 'pending' 
      });
      alert(t.uploadSuccess);
      setActiveStoryForUpload(null);
      setEpisodeForm({ title: '', file: null });
    } catch (e) { 
      alert("Error reading file"); 
    } finally { 
      setIsUploading(false); 
    }
  };

  const handleGenerateSummary = async (story: Story) => {
    if (story.episodes.length === 0) return alert(lang === 'en' ? "Add at least one episode first." : "Ongeza angalau kipindi kimoja kwanza.");
    const summary = await geminiService.generateSummary(story.episodes[0].content);
    onUpdateStory({ ...story, description: summary });
  };

  const handleGenerateAudioForEpisode = async (story: Story, episode: Episode) => {
    setIsGeneratingAudio(episode.id);
    try {
      const audio = await geminiService.generateNarration(episode.content);
      if (audio) {
        const updatedEpisodes = story.episodes.map(ep => ep.id === episode.id ? { ...ep, audioContent: audio } : ep);
        onUpdateStory({ ...story, episodes: updatedEpisodes });
      }
    } finally { setIsGeneratingAudio(null); }
  };

  return (
    <div className="max-w-6xl mx-auto pb-20">
      <div className="grid grid-cols-1 md:grid-cols-2 gap-10 items-end mb-12">
        <div>
          <h2 className="text-4xl font-serif font-bold text-slate-900">{t.creatorStudio}</h2>
          <p className="text-slate-500 mt-2 font-medium">{lang === 'en' ? 'Hello' : 'Habari'}, {currentUser.name}!</p>
        </div>
        <div className="bg-white p-8 rounded-[40px] border border-indigo-100 flex justify-between items-center shadow-sm">
          <div>
            <p className="text-[10px] font-black text-slate-400 uppercase tracking-widest mb-1">{t.earnings}</p>
            <p className="text-2xl font-black text-slate-900">TZS {balance.toLocaleString()}</p>
          </div>
          <button 
            onClick={() => setShowWithdrawModal(true)}
            className="px-6 py-3 bg-indigo-600 text-white rounded-2xl font-black text-[10px] uppercase tracking-widest hover:bg-indigo-700 transition"
          >
            {t.withdraw}
          </button>
        </div>
      </div>

      <div className="flex justify-end mb-8">
        <button onClick={() => setIsAddingProject(true)} className="bg-indigo-600 text-white px-8 py-4 rounded-2xl font-black uppercase tracking-widest shadow-xl shadow-indigo-100 hover:bg-indigo-700 hover:-translate-y-1 transition-all flex items-center gap-2">
          <svg className="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth={3} d="M12 4v16m8-8H4" /></svg>
          {t.newProject}
        </button>
      </div>

      {/* Withdraw Modal */}
      {showWithdrawModal && (
        <div className="fixed inset-0 bg-slate-900/70 backdrop-blur-md flex items-center justify-center z-[100] p-4">
          <div className="bg-white rounded-[40px] shadow-2xl p-10 max-w-lg w-full animate-scale-up border border-slate-100">
            <h3 className="text-3xl font-serif font-bold mb-8 text-slate-900">{t.withdraw}</h3>
            <div className="space-y-6">
              <div>
                <label className="block text-[10px] font-black text-slate-400 uppercase tracking-[0.2em] mb-3">{t.amountToWithdraw}</label>
                <input type="number" className="w-full px-5 py-4 rounded-2xl border border-slate-200 text-lg font-bold" value={withdrawForm.amount} onChange={e => setWithdrawForm({ ...withdrawForm, amount: parseInt(e.target.value) })} />
              </div>
              <div>
                <label className="block text-[10px] font-black text-slate-400 uppercase tracking-[0.2em] mb-3">{t.network}</label>
                <select className="w-full px-5 py-4 rounded-2xl border border-slate-200 font-bold" value={withdrawForm.network} onChange={e => setWithdrawForm({ ...withdrawForm, network: e.target.value })}>
                  <option value="M-Pesa">M-Pesa</option>
                  <option value="Tigo Pesa">Tigo Pesa</option>
                  <option value="Airtel Money">Airtel Money</option>
                </select>
              </div>
              <div>
                <label className="block text-[10px] font-black text-slate-400 uppercase tracking-[0.2em] mb-3">{t.paymentDetails}</label>
                <input type="text" className="w-full px-5 py-4 rounded-2xl border border-slate-200 text-lg font-bold" placeholder="0XXX XXXXXX" value={withdrawForm.phone} onChange={e => setWithdrawForm({ ...withdrawForm, phone: e.target.value })} />
              </div>
            </div>
            <div className="mt-10 flex gap-4">
              <button onClick={() => setShowWithdrawModal(false)} className="flex-1 px-6 py-4 border-2 border-slate-100 rounded-2xl font-black uppercase text-slate-400">{t.close}</button>
              <button onClick={handleWithdraw} className="flex-1 px-6 py-4 bg-indigo-600 text-white rounded-2xl font-black uppercase shadow-xl">{lang === 'en' ? 'Send' : 'Tuma'}</button>
            </div>
          </div>
        </div>
      )}

      {/* New Project Modal */}
      {isAddingProject && (
        <div className="fixed inset-0 bg-indigo-900/60 backdrop-blur-md flex items-center justify-center z-[100] p-4">
          <div className="bg-white rounded-[40px] shadow-2xl p-10 max-w-lg w-full animate-scale-up border border-slate-100">
            <h3 className="text-3xl font-serif font-bold mb-8 text-slate-900">{t.newProject}</h3>
            <div className="space-y-6">
              <div>
                <label className="block text-[10px] font-black text-slate-400 uppercase tracking-widest mb-3">Story Title</label>
                <input type="text" className="w-full px-5 py-4 rounded-2xl border border-slate-200 text-lg font-medium" value={projectForm.title} onChange={e => setProjectForm({...projectForm, title: e.target.value})} placeholder="e.g. Mapenzi ya Siri" />
              </div>
              <div>
                <label className="block text-[10px] font-black text-slate-400 uppercase tracking-widest mb-3">Description</label>
                <textarea className="w-full px-5 py-4 rounded-2xl border border-slate-200" rows={3} value={projectForm.description} onChange={e => setProjectForm({...projectForm, description: e.target.value})} placeholder="Briefly describe your story..." />
              </div>
              <div className="grid grid-cols-2 gap-4">
                 <div>
                    <label className="block text-[10px] font-black text-slate-400 uppercase tracking-widest mb-3">Type</label>
                    <select className="w-full px-5 py-4 rounded-2xl border border-slate-200 font-bold" value={projectForm.type} onChange={e => setProjectForm({...projectForm, type: e.target.value as any})}>
                      <option value="text">Text Story</option>
                      <option value="audio">Audio Drama</option>
                    </select>
                 </div>
                 <div>
                    <label className="block text-[10px] font-black text-slate-400 uppercase tracking-widest mb-3">Pricing</label>
                    <select className="w-full px-5 py-4 rounded-2xl border border-slate-200 font-bold" value={projectForm.isPaid ? 'paid' : 'free'} onChange={e => setProjectForm({...projectForm, isPaid: e.target.value === 'paid'})}>
                      <option value="paid">Premium (Paid)</option>
                      <option value="free">Free</option>
                    </select>
                 </div>
              </div>
            </div>
            <div className="mt-10 flex gap-4">
              <button onClick={() => setIsAddingProject(false)} className="flex-1 px-6 py-4 border-2 border-slate-100 rounded-2xl font-black uppercase text-slate-400">{t.close}</button>
              <button onClick={handleCreateStory} className="flex-1 px-6 py-4 bg-indigo-600 text-white rounded-2xl font-black uppercase shadow-xl">Create Project</button>
            </div>
          </div>
        </div>
      )}

      {/* Upload Episode Modal */}
      {activeStoryForUpload && (
        <div className="fixed inset-0 bg-slate-900/60 backdrop-blur-md flex items-center justify-center z-[100] p-4">
          <div className="bg-white rounded-[40px] shadow-2xl p-10 max-w-lg w-full animate-scale-up border border-slate-100">
            <h3 className="text-3xl font-serif font-bold mb-2 text-slate-900">{t.uploadEpisode}</h3>
            <p className="text-slate-500 mb-8">{activeStoryForUpload.title}</p>
            <div className="space-y-6">
              <div>
                <label className="block text-[10px] font-black text-slate-400 uppercase tracking-widest mb-3">{t.episodeTitle}</label>
                <input type="text" className="w-full px-5 py-4 rounded-2xl border border-slate-200 text-lg font-medium" value={episodeForm.title} onChange={e => setEpisodeForm({...episodeForm, title: e.target.value})} placeholder="e.g. Chapter 1: The Beginning" />
              </div>
              <div>
                <label className="block text-[10px] font-black text-slate-400 uppercase tracking-widest mb-3">{t.selectFile}</label>
                <div 
                  onClick={() => document.getElementById('episodeFile')?.click()}
                  className="w-full p-10 rounded-3xl border-2 border-dashed border-slate-200 flex flex-col items-center justify-center cursor-pointer hover:border-indigo-600 hover:bg-indigo-50 transition-all"
                >
                  <svg className="w-12 h-12 text-slate-300 mb-4" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M7 16a4 4 0 01-.88-7.903A5 5 0 1115.9 6L16 6a5 5 0 011 9.9M15 13l-3-3m0 0l-3 3m3-3v12" /></svg>
                  <p className="font-bold text-slate-500">{episodeForm.file ? episodeForm.file.name : t.supportedFiles}</p>
                  <input id="episodeFile" type="file" className="hidden" accept={activeStoryForUpload.type === 'text' ? '.txt' : '.mp3,.wav'} onChange={e => setEpisodeForm({...episodeForm, file: e.target.files?.[0] || null})} />
                </div>
              </div>
            </div>
            <div className="mt-10 flex gap-4">
              <button onClick={() => setActiveStoryForUpload(null)} className="flex-1 px-6 py-4 border-2 border-slate-100 rounded-2xl font-black uppercase text-slate-400">{t.close}</button>
              <button 
                onClick={handleUploadEpisode} 
                disabled={isUploading}
                className="flex-1 px-6 py-4 bg-indigo-600 text-white rounded-2xl font-black uppercase shadow-xl disabled:opacity-50"
              >
                {isUploading ? t.uploading : t.uploadEpisode}
              </button>
            </div>
          </div>
        </div>
      )}
      
      {/* Project Grid */}
      <div className="grid gap-10">
        {stories.map(story => (
          <div key={story.id} className="bg-white rounded-[40px] shadow-sm border border-slate-100 p-10 hover:shadow-xl hover:border-indigo-100 transition-all duration-500">
             <div className="flex flex-col lg:flex-row items-start gap-10">
              <div className="relative shrink-0 group">
                <img src={story.coverImage} className="w-48 h-64 rounded-[30px] object-cover shadow-2xl group-hover:scale-105 transition-transform duration-500" alt="" />
                <div className="absolute top-4 left-4">
                  <span className={`px-4 py-1.5 rounded-xl text-[10px] font-black uppercase tracking-widest shadow-xl ${
                    story.status === 'approved' ? 'bg-emerald-500 text-white' : 
                    story.status === 'pending' ? 'bg-amber-500 text-white' : 'bg-rose-500 text-white'
                  }`}>
                    {story.status}
                  </span>
                </div>
              </div>
              
              <div className="flex-1">
                <div className="flex flex-col sm:flex-row justify-between items-start gap-6 mb-6">
                  <div>
                    <h4 className="text-4xl font-serif font-bold text-slate-900 mb-2">{story.title}</h4>
                    <div className="flex items-center gap-3">
                      <span className="text-[10px] font-black text-slate-400 uppercase tracking-widest border border-slate-100 px-3 py-1 rounded-full">{story.type}</span>
                      <span className="text-[10px] font-black text-indigo-600 uppercase tracking-widest border border-indigo-50 px-3 py-1 rounded-full">{story.isPaid ? 'Premium' : 'Free'}</span>
                    </div>
                  </div>
                  <div className="flex gap-2">
                    <button onClick={() => setActiveStoryForUpload(story)} className="px-6 py-3 bg-indigo-600 text-white rounded-2xl text-xs font-black uppercase tracking-widest hover:bg-indigo-700 transition flex items-center gap-2">
                      <svg className="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth={3} d="M4 16v1a3 3 0 003 3h10a3 3 0 003-3v-1m-4-8l-4-4m0 0L8 8m4-4v12" /></svg>
                      {t.uploadEpisode}
                    </button>
                    <button onClick={() => handleGenerateSummary(story)} className="px-6 py-3 bg-slate-50 text-slate-600 rounded-2xl text-xs font-black uppercase tracking-widest hover:bg-slate-100 transition border border-slate-100">âœ¨ AI Blurb</button>
                  </div>
                </div>
                <p className="text-slate-500 text-lg leading-relaxed mb-8 max-w-3xl italic">"{story.description || (lang === 'en' ? "Your story needs a description." : "Hadithi yako inahitaji maelezo.")}"</p>
                <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 xl:grid-cols-4 gap-4">
                  {story.episodes.map(ep => (
                    <div key={ep.id} className="p-6 bg-slate-50 rounded-3xl border border-slate-100 hover:border-indigo-600 hover:bg-white transition-all group flex flex-col justify-between min-h-[160px] shadow-sm">
                      <div className="flex justify-between items-center mb-4">
                        <span className="text-[9px] font-black text-indigo-400 uppercase tracking-widest">Ep {ep.order}</span>
                      </div>
                      <h6 className="font-bold text-slate-800 line-clamp-2 mb-4">{ep.title}</h6>
                      {!ep.audioContent && story.type === 'text' ? (
                        <button disabled={isGeneratingAudio === ep.id} onClick={() => handleGenerateAudioForEpisode(story, ep)} className="w-full py-3 text-[9px] font-black uppercase tracking-widest text-indigo-600 bg-white border border-indigo-50 rounded-2xl hover:bg-indigo-600 hover:text-white transition-all">
                          {isGeneratingAudio === ep.id ? ' Narrating...' : t.addAudio}
                        </button>
                      ) : ep.audioContent || story.type === 'audio' ? <div className="text-[9px] text-emerald-600 font-black uppercase tracking-widest bg-emerald-50/50 px-4 py-2 rounded-2xl border border-emerald-100 text-center">Ready</div> : null}
                    </div>
                  ))}
                  <button onClick={() => setActiveStoryForUpload(story)} className="p-6 rounded-3xl border-2 border-dashed border-slate-100 hover:border-indigo-400 hover:bg-indigo-50/50 transition-all flex flex-col items-center justify-center text-slate-300 group min-h-[160px]">
                    <svg className="w-8 h-8 mb-2" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth={3} d="M12 4v16m8-8H4" /></svg>
                    <span className="text-[10px] font-black uppercase tracking-widest">{t.uploadEpisode}</span>
                  </button>
                </div>
              </div>
            </div>
          </div>
        ))}
      </div>
    </div>
  );
};
